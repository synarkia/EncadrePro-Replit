### STEP 3 — Extend `lib/compute-line.ts` with the full VR TN/TA formula

Open `lib/compute-line.ts`. Find the branch that handles `pricing_mode === 'square_meter'` (VR products). Replace it with the full formula below. Keep the existing `unit` and `linear_meter` branches unchanged.

**Function signature** (adjust to match current signature if it differs):

```ts
export function computeLignePvuht(input: {
  pricing_mode: 'unit' | 'linear_meter' | 'square_meter';
  product: {
    pauht: number;              // prix d'achat HT (was prix_achat_ht in some places)
    coef_marge: number;         // (was coefficient_marge)
    // VR-specific (nullable, required when pricing_mode === 'square_meter')
    majo_epaisseur?: number | null;
    mini_fact_tn?: number | null;
    mini_fact_ta?: number | null;
    coef_marge_ta?: number | null;
    plus_value_ta_pct?: number | null;
  };
  largeur_cm?: number;
  hauteur_cm?: number;
  longueur_cm?: number;
  quantite: number;
  regime?: 'TN' | 'TA';        // only used for square_meter; defaults to 'TN'
}): {
  pvuht: number;
  surface_m2?: number;
  longueur_ml?: number;
}
```

**Body for the `square_meter` branch:**

```ts
// For VR (glass, mirror, plexi): TN or TA regime
const regime = input.regime ?? 'TN';
const largeur = input.largeur_cm ?? 0;
const hauteur = input.hauteur_cm ?? 0;
const rawSurface = (largeur * hauteur) / 10000;  // cm² → m²

const majo = input.product.majo_epaisseur ?? 1;

if (regime === 'TN') {
  const miniTn = input.product.mini_fact_tn ?? 0;
  const surface = Math.max(rawSurface, miniTn);
  const coef = input.product.coef_marge;
  const pvuht = input.product.pauht * majo * coef * surface * input.quantite;
  return { pvuht, surface_m2: surface };
} else {
  // TA regime
  const miniTa = input.product.mini_fact_ta ?? 0;
  const surface = Math.max(rawSurface, miniTa);
  const coefTa = input.product.coef_marge_ta ?? input.product.coef_marge;
  const plusValue = input.product.plus_value_ta_pct ?? 0;
  const base = input.product.pauht * majo * coefTa * surface * input.quantite;
  const pvuht = base * (1 + plusValue / 100);
  return { pvuht, surface_m2: surface };
}
```
